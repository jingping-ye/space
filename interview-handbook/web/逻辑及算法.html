快速排序

function quickSort(arr) {
  if (arr.length < 2) {
    return arr;
  }
  const pivot = arr.splice(Math.floor(arr.length / 2), 1); // 选择一个基准
  const less = []; // 更小的数
  const greater = []; // 更大的数
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] <= privot) {
      less.push(arr[i]);
    } else {
      greater.push(arr[i]);
    }
  }
  return quickSort(less).concat(pivot, quickSort(greater)); // 继续调用
}
